89  // If the player has not touched the screen, we are done with this update.

93  return;

95 
96  // Should not handle input if the player is pointing on UI.

99  return;
100  }
101 
102  // If we havent set the gameplane yet
103  if (!gamePlaneSet)
104  {
105 
106  // Raycast against the location the player touched to search for planes.
107  TrackableHit hit;
108  TrackableHitFlags raycastFilter = TrackableHitFlags.PlaneWithinPolygon |
109  TrackableHitFlags.FeaturePointWithSurfaceNormal;
110 
111  if (Frame.Raycast(touch.position.x, touch.position.y, raycastFilter, out hit))
112  {
113 
114  Debug.Log("hit something");
115 
116  // Use hit pose and camera pose to check if hittest is from the
117  // back of the plane, if it is, no need to create the anchor.
118  if ((hit.Trackable is DetectedPlane) &&
119  Vector3.Dot(FirstPersonCamera.transform.position - hit.Pose.position,
120  hit.Pose.rotation * Vector3.up) < 0)
121  {
122  Debug.Log("Hit at back of the current DetectedPlane");
123  }
124  else
125  {
126  // Else, the hit is on the front of the plane
127  // Hence we make this plane the main plane of the game
128 
129  // Instantiate Hut model at the hit pose.
130  madeHut = Instantiate(hut, hit.Pose.position, hit.Pose.rotation);
131 
132  // Compensate for the hitPose rotation facing away from the raycast (i.e.
133  // camera).
134  madeHut.transform.Rotate(0, k_ModelRotation, 0, Space.Self);
135 
136  madeHut.transform.localScale *= 0.1f;
137 
138  // Create an anchor to allow ARCore to track the hitpoint as understanding of
139  // the physical world evolves.
140  var anchor = hit.Trackable.CreateAnchor(hit.Pose);
141 
142  // Make hut model a child of the anchor.
143  madeHut.transform.parent = anchor.transform;
144 
145  Debug.Log("hut position: " + madeHut.transform.position);
146 
147  Debug.Log("hit a plane");
148 
149  gamePlane = hit.Trackable;
150
